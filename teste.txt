#include <ESP8266WiFi.h>
#include <ESP8266WebServer.h>
// 
#define sinal A0
const char* ssid = "teste";
const char* password = "12345678";
const int arduinoDataPin = 0;
char* resultado;
// 
int valor;
ESP8266WebServer server(80);
void setup() {
  Serial.begin(115200);
  // 
  pinMode(sinal, INPUT);
  // 
  delay(10);
  WiFi.begin(ssid, password);
  while (WiFi.status() != WL_CONNECTED) {
    delay(250);
    Serial.flush()
    Serial.println("Conectando ao WiFi... | ");
    delay(250);
    Serial.flush()
    Serial.println("Conectando ao WiFi... / ");
    delay(250);
    Serial.flush()
    Serial.println("Conectando ao WiFi... - ");
    delay(250);
    Serial.flush()
    Serial.println("Conectando ao WiFi... \ ");
  }
  Serial.println("Conectado ao WiFi");
  Serial.print("Endere√ßo IP do ESP8266: ");
  Serial.println(WiFi.localIP());
  server.on("/", HTTP_GET, [](){
    while (Serial.available()) {
      String response = "<html lang='en'><head><style>p{font-size: 5rem}</style></head><body><p id='status'></p><script>const p=document.getElementById('status');function myLoop(){ setTimeout(function(){p.innerText=", resultado, ";myLoop();},1000);};myLoop();</script></body></html>";
    server.send(200, "text/html", response);
  });
  server.begin();
};
void loop() {
  valor = analogRead(sinal);
  if (valor > 0 && valor < 500)
  {
    Serial.println(" Status: Solo umido");
    resultado = "Status: Solo umido";
  };
  if (valor > 400 && valor < 800)
  {
    Serial.println(" Status: Umidade moderada");
    resultado = "Status: Umidade moderada";
  };
  if (valor > 800 && valor < 1024)
  {
    Serial.println(" Status: Solo seco");
    resultado = "Status: Solo seco";
  };
  delay(500);
};
